// JavaScript example
async function detectDeepfake(imageFile) {
    const formData = new FormData();
    formData.append('image', imageFile);
    formData.append('include_features', 'true');
    try {
        const response = await fetch('{{ request.url_root }}api/detect', {
            method: 'POST',
            body: formData
        });
        const result = await response.json();
        console.log('Detection result:', result);
        return result;
    } catch (error) {
        console.error('Error:', error);
    }
}

        const result = await detectDeepfake(file);
        // Handle result
    }
});
# Python example
import requests
def detect_deepfake(image_path, include_features=True):
    url = '{{ request.url_root }}api/detect'
    with open(image_path, 'rb') as image_file:
        files = {'image': image_file}
        data = {'include_features': include_features}
        response = requests.post(url, files=files, data=data)
        if response.status_code == 200:
            return response.json()
        else:
            raise Exception(f'API request failed: {response.status_code}')
# Usage
try:
    result = detect_deepfake('path/to/image.jpg')
    print(f"Result: {result['result']}")
    print(f"Confidence: {result['confidence_score']}%")
except Exception as e:
    print(f"Error: {e}")
